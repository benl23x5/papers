par-var3

H [ px -> Blocked [t1, ... tn] ] ( whnf,
 [..., px, ...],
 #px : S
 ) t

{
 (
 Blocked,
 [ ],
 Si
 ) ti }

=>
 H [ px -> (whnf[ ], [..., px, ...] ) ]
 		(
 whnf,
 [..., px, ...],
 S
 ) t

{
 (
whnf,
 [..., px, ...],
 Si
 ) ti }

where
    whnf    ->  Lambda e
                  |  Construct name a
                  |  Constant c

    i         <-    [1 .. n]
